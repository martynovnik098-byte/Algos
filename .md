//1880. Собственные числа Psych Up
#include <iostream>
#include <vector>

std::vector<int> read_numbers() {
    int n;
    std::cin >> n;
    std::vector<int> numbers(n);
    for (int i = 0; i < n; ++i) {
        std::cin >> numbers[i];
    }
    return numbers;
}

int main() {
    std::vector<int> nums1 = read_numbers();
    std::vector<int> nums2 = read_numbers();
    std::vector<int> nums3 = read_numbers();
    auto it1 = nums1.begin();
    auto it2 = nums2.begin();
    auto it3 = nums3.begin();
    int counter = 0;
    while (it1 != nums1.end() && it2 != nums2.end() && it3 != nums3.end()) {
        if (*it1 == *it2 && *it2 == *it3){
            ++counter;
            ++it1;
            ++it2;
            ++it3;
        }
        else{
            if (*it1 <= *it2 && *it1 <= *it3){
                ++it1;
            }
            else if (*it2 <= *it1 && *it2 <= *it3){
                ++it2;
            }
            else if (*it3 <= *it1 && *it3 <= *it2){
                ++it3;
            }
        }
    }
    std::cout << counter;
    return 0;
}
